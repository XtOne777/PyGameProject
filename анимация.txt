import pygame, sys

class Animation(pygame.sprite.Sprite):
    def __init__(self, pos_x, pos_y):
        super().__init__()
        self.sprites()
        self.sprites.append(pygame.image.load('./textures/scenes/frog_1.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_2.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_3.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_4.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_5.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_6.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_7.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_8.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_9.png'))
        self.sprites.append(pygame.image.load('./textures/scenes/frog_10.png'))
        self.current_sprite = 0
        self.image = self.sprites[self.current_sprite]
        self.rect = self.image.get_rect()
        self.rect.topleft = [pos_x, pos_y]

    def update(self):
        self.current_sprite += 1
        if self.current_sprite >= len(self.sprites):
            self.current_sprite = 0
        self.image = self.sprites[self.current_sprite]

pygame.init()
clock = pygame.time.Clock()

screen_width = 1200
screen_hight = 600

screen = pygame.desplay.set_mode((screen_width, screen_hight))
moving_sprites = pygame.sprite.Group()
animation = Animation(10, 10)
moving_sprites.add(animation)

while True:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
            sys.exit()

    screen.fil((0, 0, 0))
    moving_sprites.draw(screen)
    moving_sprites.update()
    pygame.display.flip()
    clock.tick(60)
